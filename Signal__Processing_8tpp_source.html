12/**
13 * @brief Applies a Finite Impulse Response (FIR) filter on the signal buffer.
14 *
15 * @tparam T Type of the buffer elements
16 * @param buffer Pointer to the input signal buffer
17 * @param kernel Pointer to the FIR filter kernel
18 * @param buf_size Size of the buffer and kernel
19 * @param kernel_sum Sum of all values in the kernel for normalization
20 * @return constexpr T Filtered output value
21 */
22template <typename T>
23constexpr T FIR_filter(T *buffer, float *kernel, int buf_size, float kernel_sum)
24{
25 if (buffer == nullptr || kernel == nullptr || buf_size <= 0 || kernel_sum == 0.0f)
26 {
27 return static_cast<T>(0);
28 }
29
30 float sum = 0;
31
32 for (int i = 0; i < buf_size; ++i)
33 {
34 sum += static_cast<float>(buffer[i]) * kernel[i];
35 }
36
37 // normalize output value
38 return static_cast<T>(sum / kernel_sum);
39}
