13 * @brief Main car status state machine.
14 *
15 * Represents the current state of the car in the startup and drive sequence.
16 */
17enum main_car_status
18{
19 INIT = 0, /**< Just started the car */
20 STARTIN = 1, /**< Driver holds "Start" button and full brakes (transition state) */

22 DRIVE = 3 /**< Ready to drive, Drive mode LED on, throttle enabled */
23};

25/**
26 * @brief Pedal fault status codes.

29 */
30enum pedal_fault_status
31{
32 NONE = 0x00, /**< No fault detected */
33 DIFF_START = 0x10, /**< >10% difference fault just started */
34 DIFF_CONTINUING = 0x11, /**< >10% difference fault is ongoing */
35 DIFF_EXCEED_100MS = 0x12, /**< >10% difference fault exceeded 100ms */
36 DIFF_RESOLVED = 0x19, /**< Difference fault resolved */
37 THROTTLE_LOW = 0x20, /**< Throttle pedal below lower threshold */
38 THROTTLE_HIGH = 0x29, /**< Throttle pedal above upper threshold */
39 BRAKE_LOW = 0x30, /**< Brake pedal below lower threshold */
40 BRAKE_HIGH = 0x39, /**< Brake pedal above upper threshold */
41};

43/**
44 * @brief BMS status

46 * Represents the current state of the Battery Management System (BMS).
47 */
48enum BMS_status
49{
50 NO_MSG = 0, /**< No message received from BMS */

53 STARTING = 3, /**< BMS is starting high voltage */
