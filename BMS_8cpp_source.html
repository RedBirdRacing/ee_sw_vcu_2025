32/**
33 * @brief Attempts to start HV.
34 * First check BMS is in standby(3) state, then send the HV start command.
35 * Keep sending the command until the BMS state changes to precharge(4).
36 * Returns when BMS state changes to run(5).
37 *
38 * @param mcp2515_ Pointer to the MCP2515 CAN controller instance.
39 */
40void BMS::check_hv(MCP2515 *mcp2515_)
41{
42 if (hv_started)
43 return; // already started
44 if (mcp2515_->readMessage(&rx_bms_msg) == MCP2515::ERROR_NOMSG)
45 {
46 DBG_BMS_STATUS(NO_MSG);
47 hv_started = false;
48 return;
49 }
50
51 // Check if the BMS is in standby state (0x3 in upper 4 bits)
52 if (rx_bms_msg.can_id != BMS_INFO_EXT)
53 {
54 DBG_BMS_STATUS(WRONG_ID);
55 hv_started = false;
56 return;
57 } // Not a BMS info frame, retry
58
59 switch (rx_bms_msg.data[6] & 0xF0)
60 {
61 case 0x30: // Standby state
62 DBG_BMS_STATUS(WAITING);
63 mcp2515_->sendMessage(&start_hv_msg);
64 DBGLN_GENERAL("BMS in standby state, sent start HV cmd");
65 // sent start HV cmd, wait for BMS to change state
66 hv_started = false;
67 return;
68 case 0x40: // Precharge state
69 DBG_BMS_STATUS(STARTING);
70 mcp2515_->sendMessage(&start_hv_msg);
71 DBGLN_GENERAL("BMS in precharge state, HV starting");
72 hv_started = false;
73 return; // BMS is in precharge state, wait
74 case 0x50: // Run state
75 DBG_BMS_STATUS(STARTED);
76 DBGLN_GENERAL("BMS in run state, HV started");
77 hv_started = true; // BMS is in run state
78 return;
79 default:
80 DBG_BMS_STATUS(UNUSED);
81 DBGLN_GENERAL("BMS in unknown state, retrying...");
82 hv_started = false;
83 return; // Unknown state, retry
84 }
85 hv_started = false; // guard, probably will be optimized out
86}
