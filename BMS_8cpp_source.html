92void BMS::check_hv(MCP2515 *mcp2515_)
93{
94 if (hv_started)
95 return; // already started
96 if (mcp2515_->readMessage(&rx_bms_msg) == MCP2515::ERROR_NOMSG)
97 {
98 DBG_BMS_STATUS(NO_MSG);
99 hv_started = false;
100 return;
101 }
102
103 // Check if the BMS is in standby state (0x3 in upper 4 bits)
104 if (rx_bms_msg.can_id != BMS_INFO_EXT)
105 {
106 DBG_BMS_STATUS(WRONG_ID);
107 hv_started = false;
108 return;
109 } // Not a BMS info frame, retry
110
111 switch (rx_bms_msg.data[6] & 0xF0)
112 {
113 case 0x30: // Standby state
114 DBG_BMS_STATUS(WAITING);
115 hv_send_start = true;
116 // last_msg_ms = *current_ms;
117 DBGLN_GENERAL("BMS in standby state, sent start HV cmd");
118 // sent start HV cmd, wait for BMS to change state
119 hv_started = false;
120 return;
121 case 0x40: // Precharge state
122 DBG_BMS_STATUS(STARTING);
123 hv_send_start = false;
124 DBGLN_GENERAL("BMS in precharge state, HV starting");
125 hv_started = false;
126 return; // BMS is in precharge state, wait
127 case 0x50: // Run state
128 DBG_BMS_STATUS(STARTED);
129 hv_send_start = false;
130 DBGLN_GENERAL("BMS in run state, HV started");
131 hv_started = true; // BMS is in run state
132 return;
133 default:
134 DBG_BMS_STATUS(UNUSED);
135 DBGLN_GENERAL("BMS in unknown state, retrying...");
136 hv_started = false;
137 return; // Unknown state, retry
138 }
139 hv_started = false; // guard, probably will be optimized out
140}
