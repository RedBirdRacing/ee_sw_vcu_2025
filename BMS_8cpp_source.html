38/**
39 * @brief Attempts to start HV.
40 * First check BMS is in standby(3) state, then send the HV start command.
41 * Keep sending the command until the BMS state changes to precharge(4).
42 * Sets car.pedal.status.bits.hv_ready to true when BMS state changes to run(5).
43 *
44 */
45void BMS::checkHv()
46{
47 if (car.pedal.status.bits.hv_ready)
48 return; // already started
49 car.pedal.status.bits.hv_ready = false;
50 car.pedal.status.bits.bms_no_msg = false;
51 car.pedal.status.bits.bms_wrong_id = false;
52 if (bms_can.readMessage(&rx_bms_msg) == MCP2515::ERROR_NOMSG)
53 {
54 DBG_BMS_STATUS(BmsStatus::NoMsg);
55 car.pedal.status.bits.bms_no_msg = true;
56 return;
57 }
58
59 // Check if the BMS is in standby state (0x3 in upper 4 bits)
60 if (rx_bms_msg.can_id != BMS_INFO_EXT)
61 {
62 DBG_BMS_STATUS(BmsStatus::WrongId);
63 car.pedal.status.bits.bms_wrong_id = true;
64 return;
65 } // Not a BMS info frame, retry
66
67 switch (rx_bms_msg.data[6] & 0xF0)
68 {
69 case 0x30: // Standby state
70 DBG_BMS_STATUS(BmsStatus::Waiting);
71 bms_can.sendMessage(&start_hv_msg);
72 DBGLN_GENERAL("BMS in standby state, sent start HV cmd");
73 // sent start HV cmd, wait for BMS to change state
74 return;
75 case 0x40: // Precharge state
76 DBG_BMS_STATUS(BmsStatus::Starting);
77 bms_can.sendMessage(&start_hv_msg);
78 DBGLN_GENERAL("BMS in precharge state, HV starting");
79 return; // BMS is in precharge state, wait
80 case 0x50: // Run state
81 DBG_BMS_STATUS(BmsStatus::Started);
82 DBGLN_GENERAL("BMS in run state, HV started");
83 car.pedal.status.bits.hv_ready = true; // BMS is in run state
84 return;
85 default:
86 DBG_BMS_STATUS(BmsStatus::Unused);
87 DBGLN_GENERAL("BMS in unknown state, retrying...");
88 return; // Unknown state, retry
89 }
90}
