15/**
16 * @brief Main car status state machine.
17 *
18 * Represents the current state of the car in the startup and drive sequence.
19 */
20enum class CarStatus : uint8_t
21{
22 Init = 0, /**< Just started the car */
23 Startin = 1, /**< Driver holds "Start" button and full brakes (transition state) */
24 Bussin = 2, /**< Buzzer active, driver can release "Start" and brakes (transition state) */
25 Drive = 3 /**< Ready to drive, Drive mode LED on, throttle enabled */
26};

28/**
29 * @brief Pedal fault status codes.

32 */
33enum class PedalFault : uint8_t
34{
35 None = 0x00, /**< No fault detected */

37 DiffContinuing = 0x11, /**< >10% difference fault is ongoing */

39 DiffResolved = 0x19, /**< Difference fault resolved */
40 ThrottleLow = 0x20, /**< Throttle pedal below lower threshold */
41 ThrottleHigh = 0x29, /**< Throttle pedal above upper threshold */
42 BrakeLow = 0x30, /**< Brake pedal below lower threshold */
43 BrakeHigh = 0x39, /**< Brake pedal above upper threshold */
44};
