33/**
34 * @brief Sends a debug throttle input message over CAN.
35 *
36 * @param pedal_filtered_1 Filtered value from pedal sensor 1.
37 * @param pedal_filtered_2 Filtered value from pedal sensor 2.
38 * @param pedal_2_scaled Scaled value of pedal sensor 2.
39 * @param brake Brake pedal value.
40 */
41void Debug_CAN::throttle_in(uint16_t pedal_filtered_1, uint16_t pedal_filtered_2, uint16_t pedal_2_scaled, uint16_t brake)
42{
43 if (!can_interface)
44 return;
45
46 can_frame tx_msg;
47 tx_msg.can_id = THROTTLE_IN_MSG;
48 tx_msg.can_dlc = 8;
49
50 // Little-endian format for uint16_t values
51 tx_msg.data[0] = pedal_filtered_1 & 0xFF;
52 tx_msg.data[1] = (pedal_filtered_1 >> 8) & 0xFF; // Upper byte
53 tx_msg.data[2] = pedal_filtered_2 & 0xFF;
54 tx_msg.data[3] = (pedal_filtered_2 >> 8) & 0xFF; // Upper byte
55 tx_msg.data[4] = pedal_2_scaled & 0xFF;
56 tx_msg.data[5] = (pedal_2_scaled >> 8) & 0xFF; // Upper byte
57 tx_msg.data[6] = brake & 0xFF;
58 tx_msg.data[7] = (brake >> 8) & 0xFF; // Upper byte
59
60 can_interface->sendMessage(&tx_msg);
61}

87/**
88 * @brief Sends a debug throttle fault message over CAN with a float value.
89 *
90 * @param fault_status The status of the throttle fault as defined in PedalFault enum.
91 * @param value Optional uint16_t value associated with the fault (e.g., pedal voltage).
92 */
93void Debug_CAN::throttle_fault(PedalFault fault_status, uint16_t value)
94{
95 if (!can_interface)
96 return;
97
98 can_frame tx_msg;
99 tx_msg.can_id = THROTTLE_FAULT_MSG;
100 tx_msg.can_dlc = 3;
101
102 tx_msg.data[0] = static_cast<uint8_t>(fault_status); // Convert enum to uint8_t
103
104 tx_msg.data[1] = value & 0xFF;
105 tx_msg.data[2] = (value >> 8) & 0xFF; // Upper byte
106
107 can_interface->sendMessage(&tx_msg);
108}
