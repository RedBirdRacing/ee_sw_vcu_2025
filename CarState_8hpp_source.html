22/**
23 * @brief Telemetry frame structure for ADC readings.

25struct TelemetryFrameAdc
26{
27 uint16_t apps_5v; /**< ADC reading for 5V APPS */
28 uint16_t apps_3v3; /**< ADC reading for 3.3V APPS */
29 uint16_t brake; /**< ADC reading for brake pedal */

31 /**
32 * @brief Converts the TelemetryFrameAdc to a CAN frame.
33 * @return CAN frame representing the telemetry ADC readings.
34 */
35 constexpr can_frame toCanFrame() const
36 {
37 return can_frame{
38 TELEMETRY_ADC_MSG, // can_id
39 8, // can_dlc
40 static_cast<__u8>(apps_5v & 0xFF), // data
41 static_cast<__u8>((apps_5v >> 8) & 0xFF),
42 static_cast<__u8>(apps_3v3 & 0xFF),
43 static_cast<__u8>((apps_3v3 >> 8) & 0xFF),
44 static_cast<__u8>(brake & 0xFF),
45 static_cast<__u8>((brake >> 8) & 0xFF),
46 static_cast<__u8>(hall_sensor & 0xFF),
47 static_cast<__u8>((hall_sensor >> 8) & 0xFF)};
48 }

81/**
82 * @brief Telemetry frame structure for VCU states.
83 */
84struct TelemetryFrameState
85{
86 union StateByte0
87 {
88 uint8_t byte;
89 struct
90 {
91 CarStatus car_status : 2; /**< Current car status */
92 bool state_unknown : 1; /**< Unknown car state */
93 bool hv_ready : 1; /**< High voltage ready */
94 bool bms_no_msg : 1; /**< BMS read no message */
95 bool bms_wrong_id : 1; /**< BMS read wrong ID */
96 bool force_stop : 1; /**< Fault forced car to stop */
97 bool screenshot : 1; /**< Screenshot, throttle + brake > threshold */
98 } bits;
99 };
100
101 union StateByte1
102 {
103 uint8_t byte;
104 struct
105 {
106 bool fault_active : 1; /**< Pedal faulty now */
107 bool fault_exceeded : 1; /**< Current pedal fault exceeded allowed time */
108 bool apps_5v_low : 1; /**< APPS 5V considered shorted to ground*/
109 bool apps_5v_high : 1; /**< APPS 5V considered shorted to rail */
110 bool apps_3v3_low : 1; /**< APPS 3V3 considered shorted to ground */
111 bool apps_3v3_high : 1; /**< APPS 3V3 considered shorted to rail */
112 bool brake_low : 1; /**< Brake considered shorted to ground */
113 bool brake_high : 1; /**< Brake considered shorted to rail */
114 } bits;
115 };
116
117 static_assert(sizeof(StateByte0) == 1, "TelemetryStateByte0 must be 1 byte"); // ensure compile is shoving the bits as expected
118 static_assert(sizeof(StateByte1) == 1, "TelemetryStateByte1 must be 1 byte");
119
120 StateByte0 status; /**< First status byte */
121 StateByte1 faults; /**< Second status byte */
122
123 uint8_t bms_data[6]; /**< Raw BMS data bytes */
124
125 /**
126 * @brief Converts the TelemetryFrameState to a CAN frame.
127 * @return CAN frame representing the telemetry car state.
128 */
129 constexpr can_frame toCanFrame() const
130 {
131 return can_frame{
132 TELEMETRY_STATE_MSG, // can_id
133 8, // can_dlc
134 status.byte,
135 faults.byte,
136 bms_data[0],
137 bms_data[1],
138 bms_data[2],
139 bms_data[3],
140 bms_data[4],
141 bms_data[5]};
142 }
143};
