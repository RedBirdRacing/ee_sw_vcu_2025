35 union StateByteStatus /**< Union of bits for car status besides Pedal */
36 {
37 uint8_t byte;
38 struct Bits
39 {
40 CarStatus car_status : 2; /**< Current car status, produces compiler warning before GCC 9.3 due to bug */
41 bool state_unknown : 1; /**< Unknown car state */
42 bool hv_ready : 1; /**< High voltage ready */
43 bool bms_no_msg : 1; /**< BMS read no message */
44 bool bms_wrong_id : 1; /**< BMS read wrong ID */
45 bool screenshot : 1; /**< Screenshot, throttle + brake > threshold */
46 bool force_stop : 1; /**< Fault forced car to stop */
47 } bits;
48 };
49 union StateByteFaults /**< Union of bits for pedal faults */
50 {
51 uint8_t byte;
52 struct Bits
53 {
54 bool fault_active : 1; /**< Pedal faulty now, only one resetable */
55 bool fault_exceeded : 1; /**< Current pedal fault exceeded allowed time */
56 bool apps_5v_low : 1; /**< APPS 5V considered shorted to ground*/
57 bool apps_5v_high : 1; /**< APPS 5V considered shorted to rail */
58 bool apps_3v3_low : 1; /**< APPS 3V3 considered shorted to ground */
59 bool apps_3v3_high : 1; /**< APPS 3V3 considered shorted to rail */
60 bool brake_low : 1; /**< Brake considered shorted to ground */
61 bool brake_high : 1; /**< Brake considered shorted to rail */
62 } bits;
63 };

71 /**
72 * @brief Converts the TelemetryFramePedal to a CAN frame.
73 * @return CAN frame representing the Pedal telemetry signals.
74 */
75 constexpr can_frame toCanFrame() const
76 {
77 return can_frame{
78 TELEMETRY_PEDAL_MSG, // can_id
79 8, // can_dlc
80 static_cast<__u8>(apps_5v & 0xFF), // data
81 static_cast<__u8>(((apps_5v >> 8) & 0x03) | ((apps_3v3 & 0x3F) << 2)),
82 static_cast<__u8>(((apps_3v3 >> 6) & 0x0F) | ((brake & 0x0F) << 4)),
83 static_cast<__u8>(((brake >> 4) & 0x3F) | ((hall_sensor & 0x03) << 6)),
84 static_cast<__u8>((hall_sensor >> 2) & 0xFF),
85 status.byte,
86 faults.byte,
87 0x00};
88 }
89};
