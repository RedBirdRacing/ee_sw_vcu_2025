77 * This function formats and sends the throttle fault status and value to the serial console.
78 *
79 * @param fault_status The status of the throttle fault as defined in PedalFault enum.
80 * @param value Optional uint16_t value associated with the fault

82void Debug_Serial::throttle_fault(PedalFault fault_status, uint16_t value)
83{
84 switch (fault_status)
85 {
86 case PedalFault::None:
87 break;
88 case PedalFault::DiffContinuing:
89 Serial.print("Pedal mismatch continuing. Difference: ");
90 Serial.println(value);
91 break;
92 case PedalFault::ThrottleLow:
93 Serial.print("Throttle input too low. Value: ");
94 Serial.println(value);
95 break;
96 case PedalFault::ThrottleHigh:
97 Serial.print("Throttle too high. Value: ");
98 Serial.println(value);
99 break;
100 default:
101 Serial.println("Unknown fault status");
102 break;
103 }
104}

166void Debug_Serial::status_car(CarStatus car_status)

169 if (car_status != last_status) {
170 Serial.print("Car Status: ");
171 switch (last_status)

173 case CarStatus::Init:
174 Serial.print("INIT -> ");
175 break;
176 case CarStatus::Startin:
177 Serial.print("STARTIN -> ");
178 break;
179 case CarStatus::Bussin:
180 Serial.print("BUSSIN -> ");
181 break;
182 case CarStatus::Drive:
183 Serial.print("DRIVE -> ");
184 break;
185 default:
186 Serial.print("UNKNOWN -> ");
187 break;
188 }
189 switch (car_status)
190 {
191 case CarStatus::Init:
192 Serial.print("INIT");
193 break;
194 case CarStatus::Startin:
195 Serial.print("STARTIN");
196 break;
197 case CarStatus::Bussin:
198 Serial.print("BUSSIN");
199 break;
200 case CarStatus::Drive:
201 Serial.print("DRIVE");
202 break;
203 default:
204 Serial.print("UNKNOWN");
205 break;
206 }
207 last_status = car_status;
208 }
209}
