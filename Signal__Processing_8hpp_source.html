47/**
48 * @brief Filter with exponential moving average algorithm.
49 * f(t) = (f(t-1) * OLD_RATIO + sample * NEW_RATIO + (OLD_RATIO + NEW_RATIO) / 2) / (OLD_RATIO + NEW_RATIO)
50 * Due to round down, won't ever reach maximum, especially if OLD_RATIO >> NEW_RATIO, so use of curve important.
51 * @tparam TypeInput Type of the input samples.
52 * @tparam TypeMid Type used for intermediate calculations.
53 * @tparam OLD_RATIO Weighting ratio for the old value.
54 * @tparam NEW_RATIO Weighting ratio for the new sample.
55 */
56template <typename TypeInput, typename TypeMid, uint8_t OLD_RATIO = 31, uint8_t NEW_RATIO = 1>
57class ExponentialFilter : public Filter<TypeInput, TypeMid>
58{
59public:
60 ExponentialFilter();
61 void addSample(TypeInput sample) override;
62 TypeInput getFiltered() const override;
63
64private:
65 TypeInput last_out = 0; /**< Last output value for exponential filter, input for next calculation */
66};
