18/**
19 * @brief Telemetry frame structure for ADC readings.
20 */
21struct TelemetryFrameAdc
22{
23 uint16_t apps_5v; /**< ADC reading for 5V APPS */
24 uint16_t apps_3v3; /**< ADC reading for 3.3V APPS */
25 uint16_t brake; /**< ADC reading for brake pedal */
26 uint16_t hall_sensor; /**< ADC reading for hall sensor */
27 /**
28 * @brief Converts the TelemetryFrameAdc to a CAN frame.
29 * @return CAN frame representing the telemetry ADC readings.
30 */
31 constexpr can_frame toCanFrame() const
32 {
33 return can_frame{
34 TELEMETRY_ADC_MSG, // can_id
35 8, // can_dlc
36 static_cast<__u8>(apps_5v & 0xFF), // data
37 static_cast<__u8>((apps_5v >> 8) & 0xFF),
38 static_cast<__u8>(apps_3v3 & 0xFF),
39 static_cast<__u8>((apps_3v3 >> 8) & 0xFF),
40 static_cast<__u8>(brake & 0xFF),
41 static_cast<__u8>((brake >> 8) & 0xFF),
42 static_cast<__u8>(hall_sensor & 0xFF),
43 static_cast<__u8>((hall_sensor >> 8) & 0xFF)};
44 }
45};

77/**
78 * @brief Telemetry frame structure for VCU states.
79 */
80struct TelemetryFrameState
81{
82 // first byte
83
84 main_car_status car_status; /**< Current car status */
85 bool force_stop; /**< Fault forced car to stop */
86 bool hv_ready; /**< High voltage ready */
87 bool bms_no_msg; /**< BMS read no message */
88 bool bms_wrong_id; /**< BMS read wrong ID */
89
90 // second byte
91 bool fault_active; /**< Pedal faulty now */
92 bool fault_exceeded; /**< Current pedal fault exceeded allowed time */
93 bool throttle_low; /**< Throttle lower than minimum */
94 bool throttle_high; /**< Throttle higher than maximum */
95 bool brake_low; /**< Brake lower than minimum */
96 bool brake_high; /**< Brake higher than maximum */
97 bool screenshot; /**< Screenshot, throttle + brake > threshold */
98 bool state_unknown; /**< CarState unknown, consider as bit-flip */
99
100 uint8_t bms_data[6]; /**< Raw BMS data bytes */
101
102 /**
103 * @brief Converts the TelemetryFrameState to a CAN frame.
104 * @return CAN frame representing the telemetry car state.
105 */
106 constexpr can_frame toCanFrame() const
107 {
108 return can_frame{
109 TELEMETRY_STATE_MSG, // can_id
110 8, // can_dlc
111 static_cast<__u8>(
112 (static_cast<uint8_t>(car_status) & 0x0F) |
113 (force_stop ? 0x10 : 0x00) |
114 (hv_ready ? 0x20 : 0x00) |
115 (bms_no_msg ? 0x40 : 0x00) |
116 (bms_wrong_id ? 0x80 : 0x00)), // data
117 static_cast<__u8>(
118 (fault_active ? 0x01 : 0x00) |
119 (fault_exceeded ? 0x02 : 0x00) |
120 (throttle_low ? 0x04 : 0x00) |
121 (throttle_high ? 0x08 : 0x00) |
122 (brake_low ? 0x10 : 0x00) |
123 (brake_high ? 0x20 : 0x00) |
124 (screenshot ? 0x40 : 0x00) |
125 (state_unknown ? 0x80 : 0x00)),
126 bms_data[0],
127 bms_data[1],
128 bms_data[2],
129 bms_data[3],
130 bms_data[4],
131 bms_data[5]};
132 }
133};
