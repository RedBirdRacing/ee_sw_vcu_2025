37/**
38 * @brief Updates pedal sensor readings, applies filtering, and checks for faults.
39 *
40 * Stores new pedal readings, applies an average filter, and updates car state.
41 * If a fault is detected between pedal sensors, sets fault flags and logs status.
42 *
43 * @param pedal_1 Raw value from pedal sensor 1.
44 * @param pedal_2 Raw value from pedal sensor 2.
45 * @param brake Raw value from brake sensor.
46 */
47void Pedal::update(uint16_t pedal_1, uint16_t pedal_2, uint16_t brake)
48{
49 // Record readings in buffer
50 pedal_value_1.push(pedal_1);
51 pedal_value_2.push(pedal_2);
52 brake_value.push(brake);
53
54 // Range of pedal 1 is APPS_PEDAL_1_RANGE, pedal 2 is APPS_PEDAL_2_RANGE;
55
56 // this is current taking the direct array the circular queue writes into. Bad idea to do anything other than a simple average
57 // if not using a linear filter, pass the pedalValue_1.getLinearBuffer() to the filter function to ensure the ordering is correct.
58 // can also consider injecting the filter into the queue if need
59 // depends on the hardware filter, reduce software filtering as much as possible
60
61 // currently a small average filter is good enough
62 car.adc.apps_5v = AVG_filter<uint16_t>(pedal_value_1.buffer, ADC_BUFFER_SIZE);
63 car.adc.apps_3v3 = AVG_filter<uint16_t>(pedal_value_2.buffer, ADC_BUFFER_SIZE);
64 car.adc.brake = AVG_filter<uint16_t>(brake_value.buffer, ADC_BUFFER_SIZE);
65
66 if (!checkPedalFault())
67 {
68 if (fault)
69 {
70 DBG_THROTTLE_FAULT(DIFF_RESOLVED);
71 fault = false;
72 car.state.fault_exceeded = false; // exceed flag can show multiple instances of fault exceed, so need reset
73 }
74 return;
75 }
76
77 // Pedal fault detected
78 if (fault) // if previously faulty
79 { // Previous scan is already faulty
80 if (car.millis - fault_start_millis > 100) // Faulty for more than 100 ms
81 {
82 car.state.fault_exceeded = true; // Turning off the motor is achieved using another digital pin, not via canbus, but will still send 0 torque can frames
83 car.state.force_stop = true; // this force stop flag can only be reset by a power cycle
84
85 DBG_THROTTLE_FAULT(DIFF_EXCEED_100MS);
86 return;
87 }
88 }
89 else // new fault detected
90 {
91 fault_start_millis = car.millis;
92 DBG_THROTTLE_FAULT(DIFF_START);
93 }
94
95 fault = true;
96 return;
97}

100 * @brief Sends the appropriate CAN frame based on pedal and car state.

102void Pedal::sendFrame()
103{
104 if (car.state.force_stop)
105 {
106 DBGLN_THROTTLE("Stopping motor: pedal fault");
107 motor_can.sendMessage(&stop_frame);
108 return;
109 }
110 if (car.state.car_status != DRIVE)
111 {
112 switch (car.state.car_status)
113 {
114 case INIT:
115 DBGLN_THROTTLE("Stopping motor: in INIT.");
116 break;
117 case STARTIN:
118 DBGLN_THROTTLE("Stopping motor: in STARTIN.");
119 break;
120 case BUSSIN:
121 DBGLN_THROTTLE("Stopping motor: in BUSSIN.");
122 break;
123 default:
124 DBGLN_THROTTLE("Stopping motor: in UNKNOWN STATE.");
125 break;
126 }
127 motor_can.sendMessage(&stop_frame);
128 return;
129 }
130
131 int16_t torque_val = throttleTorqueMapping(pedal_final, car.adc.brake, FLIP_MOTOR_DIR);
132
133 DBG_THROTTLE_OUT(pedal_final, torque_val);
134
135 torque_msg.can_id = MOTOR_COMMAND;
136 torque_msg.can_dlc = 3;
137 torque_msg.data[0] = 0x90; // 0x90 for torque, 0x31 for speed
138 torque_msg.data[1] = torque_val & 0xFF;
139 torque_msg.data[2] = (torque_val >> 8) & 0xFF;
140 motor_can.sendMessage(&torque_msg);
141 return;
142}

172/**
173 * @brief Checks for a fault between two pedal sensor readings.
174 *
175 * Scales pedal_2 to match the range of pedal_1, then calculates the absolute difference.
176 * If the difference exceeds 10% of the full-scale value (i.e., >102.4 for a 10-bit ADC),
177 * the function considers this a fault and returns true. Otherwise, returns false.
178 *
179 * @return true if the difference exceeds the threshold (fault detected), false otherwise.
180 */
181bool Pedal::checkPedalFault()
182{
183 car.digital.apps_3v3_scaled = car.adc.apps_3v3 * APPS_RATIO;
184 DBG_THROTTLE_IN(car.adc.apps_5v, car.adc.apps_3v3, car.digital.apps_3v3_scaled, car.adc.brake);
185
186 int16_t delta = (int16_t)car.adc.apps_5v - (int16_t)car.digital.apps_3v3_scaled;
187 // if more than 10% difference between the two pedals, consider it a fault
188 if (delta > 102.4 || delta < -102.4) // 10% of 1024, rounded down to 102
189 {
190 car.state.fault_active = true;
191 DBG_THROTTLE_FAULT(DIFF_CONTINUING, delta);
192 return true;
193 }
194 car.state.fault_active = false;
195 return false;
196}
