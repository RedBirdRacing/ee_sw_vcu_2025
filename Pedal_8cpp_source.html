36/**
37 * @brief Updates pedal sensor readings, applies filtering, and checks for faults.
38 *
39 * Stores new pedal readings, applies an average filter, and updates car state.
40 * If a fault is detected between pedal sensors, sets fault flags and logs status.
41 *
42 * @param car Pointer to car_state structure to update.
43 * @param pedal_1 Raw value from pedal sensor 1.
44 * @param pedal_2 Raw value from pedal sensor 2.
45 * @param brake Raw value from brake sensor.
46 */
47void Pedal::pedal_update(car_state *car, uint16_t pedal_1, uint16_t pedal_2, uint16_t brake)
48{
49 // Record readings in buffer
50 pedal_value_1.push(pedal_1);
51 pedal_value_2.push(pedal_2);
52 brake_value.push(brake);
53
54 // Range of pedal 1 is APPS_PEDAL_1_RANGE, pedal 2 is APPS_PEDAL_2_RANGE;
55
56 // this is current taking the direct array the circular queue writes into. Bad idea to do anything other than a simple average
57 // if not using a linear filter, pass the pedalValue_1.getLinearBuffer() to the filter function to ensure the ordering is correct.
58 // can also consider injecting the filter into the queue if need
59 // depends on the hardware filter, reduce software filtering as much as possible
60
61 // currently a small average filter is good enough
62 pedal_filtered_1 = AVG_filter<uint16_t>(pedal_value_1.buffer, ADC_BUFFER_SIZE);
63 pedal_filtered_2 = AVG_filter<uint16_t>(pedal_value_2.buffer, ADC_BUFFER_SIZE);
64 car->brake_final = AVG_filter<uint16_t>(brake_value.buffer, ADC_BUFFER_SIZE);
65
66 car->pedal_final = pedal_filtered_1; // Only take in pedal 1 value
67
68 if (!check_pedal_fault(pedal_filtered_1, pedal_filtered_2, car->brake_final))
69 {
70 if (fault)
71 {
72 DBG_THROTTLE_FAULT(DIFF_RESOLVED);
73 fault = false;
74 }
75 return;
76 }
77
78 // Pedal fault detected
79 if (fault) // if previously faulty
80 { // Previous scan is already faulty
81 if (car->millis - fault_start_millis > 100) // Faulty for more than 100 ms
82 {
83 // Turning off the motor is achieved using another digital pin, not via canbus, but will still send 0 torque can signals
84 car->fault_force_stop = true;
85 // this force stop flag can only be reset by a power cycle
86
87 DBG_THROTTLE_FAULT(DIFF_EXCEED_100MS);
88 return;
89 }
90 }
91 else // new fault detected
92 {
93 fault_start_millis = car->millis;
94 DBG_THROTTLE_FAULT(DIFF_START);
95 }
96
97 fault = true;
98 return;
99}

120/**
121 * @brief Updates the CAN frame with the most recent pedal value.
122 *
123 * This function prepares a CAN frame to send the current pedal value to the motor controller.
124 * It checks for faults and applies appropriate torque values based on pedal readings.
125 *
126 * @param tx_throttle_msg Pointer to the CAN frame to update.
127 * @param car Pointer to the car_state structure containing current car state.
128 */
129void Pedal::pedal_can_frame_update(can_frame *tx_throttle_msg, const car_state *car)
130{
131 if (tx_throttle_msg == nullptr || car == nullptr)
132 return;
133 if (car->fault_force_stop)
134 {
135 pedal_can_frame_stop_motor(tx_throttle_msg);
136 DBGLN_THROTTLE("Stopping motor: pedal fault");
137 return;
138 }
139 if (car->car_status != DRIVE)
140 {
141 pedal_can_frame_stop_motor(tx_throttle_msg);
142 switch (car->car_status)
143 {
144 case INIT:
145 DBGLN_THROTTLE("Stopping motor: in INIT.");
146 break;
147 case STARTIN:
148 DBGLN_THROTTLE("Stopping motor: in STARTIN.");
149 break;
150 case BUSSIN:
151 DBGLN_THROTTLE("Stopping motor: in BUSSIN.");
152 break;
153 default:
154 DBGLN_THROTTLE("Stopping motor: in UNKNOWN STATE.");
155 break;
156 }
157 return;
158 }
159
160 int16_t throttle_torque_val = throttle_torque_mapping(car->pedal_final, car->brake_final, FLIP_MOTOR_DIR);
161
162 DBG_THROTTLE_OUT(car->pedal_final, throttle_torque_val);
163
164 tx_throttle_msg->can_id = MOTOR_COMMAND;
165 tx_throttle_msg->can_dlc = 3;
166 tx_throttle_msg->data[0] = 0x90; // 0x90 for torque, 0x31 for speed
167 tx_throttle_msg->data[1] = throttle_torque_val & 0xFF;
168 tx_throttle_msg->data[2] = (throttle_torque_val >> 8) & 0xFF;
169}

171/**
172 * @brief Maps the pedal ADC to a torque value, handling deadzones and faults.
173 *
174 * This function takes a pedal ADC in the range of 0-1023 and maps it to a torque value.
175 * It handles deadzones, faults, and applies the mapping or returns fixed values as needed.
176 *
177 * @param pedal Pedal ADC in the range of 0-1023.
178 * @param brake Brake ADC in the range of 0-1023.
179 * @param flip_dir Boolean indicating whether to flip the motor direction.
180 * @return Mapped torque value in the signed range of -32760 to 32760.
181 */
182int16_t Pedal::throttle_torque_mapping(uint16_t pedal, uint16_t brake, bool flip_dir)
183{
184 if (pedal < PEDAL_LL)
185 {
186 DBG_THROTTLE_FAULT(THROTTLE_LOW, pedal);
187 return 0;
188 }
189 else if (pedal < PEDAL_LU)
190 {
191// in lower deadzone, treat as 0% throttle, can regen
192#if REGEN_ENABLED
193 return brake_torque_mapping(brake, flip_dir);
194#else
195 return 0;
196#endif
197 }
198 else if (pedal < PEDAL_UL)
199 {
200 // throttle_in -> torque_val
201 // temp linear mapping, with proper casting to prevent overflow
202 int32_t numerator = static_cast<int32_t>(pedal - PEDAL_1_LU) * static_cast<int32_t>(MAX_THROTTLE_OUT_VAL);
203 int32_t denominator = static_cast<int32_t>(PEDAL_1_UL - PEDAL_1_LU);
204 int16_t result = static_cast<int16_t>(numerator / denominator);
205
206 if (flip_dir)
207 return -result;
208 return result;
209 }
210 else if (pedal < PEDAL_UU)
211 {
212 // in upper deadzone, treat as 100% throttle
213 if (flip_dir)
214 return -MAX_THROTTLE_OUT_VAL;
215 else
216 return MAX_THROTTLE_OUT_VAL;
217 }
218 else
219 {
220 // throttle higher than upper deadzone, treat as throttle fault, zeroing torque for safety
221 DBG_THROTTLE_FAULT(THROTTLE_HIGH, pedal);
222 return 0;
223 }
224}

226/**
227 * @brief Maps the brake ADC to a torque value.
228 *
229 * This function takes a brake ADC in the range of 0-1023 and maps it to a torque value.
230 * The mapping is linear, and the result is adjusted based on the motor direction.
231 *
232 * @param brake Brake ADC in the range of 0-1023.
233 * @param flip_dir Boolean indicating whether to flip the motor direction.
234 * @return Mapped torque value in the signed range of -32760 to 32760.
235 */
236int16_t Pedal::brake_torque_mapping(uint16_t brake, bool flip_dir)
237{
238 int16_t result = 0;
239 if (brake < BRAKE_LL)
240 {
241 DBG_BRAKE_FAULT(BRAKE_LOW, brake);
242 return 0; // no need mess with result, directly return
243 }
244 else if (brake < BRAKE_LU)
245 {
246 // in lower deadzone, treat as 0% throttle, can regen
247 result = MIN_REGEN;
248 }
249 else if (brake < BRAKE_UL)
250 {
251 // throttle_in -> torque_val
252 // linear mapping, with proper casting to prevent overflow
253 int32_t numerator = static_cast<int32_t>(brake - BRAKE_LU) * static_cast<int32_t>(MAX_THROTTLE_OUT_VAL);
254 int32_t denominator = static_cast<int32_t>(BRAKE_UL - BRAKE_LU);
255 result = static_cast<int16_t>(numerator / denominator);
256 }
257 else if (brake < BRAKE_UU)
258 {
259 // in upper deadzone, treat as 100% throttle
260 result = MAX_THROTTLE_OUT_VAL;
261 }
262 else
263 {
264 // throttle higher than upper deadzone, treat as fault, zeroing for safety
265 DBG_THROTTLE_FAULT(BRAKE_HIGH, brake);
266 return 0;
267 }
268 if (flip_dir)
269 return -result;
270 return result;
271}

273/**
274 * @brief Checks for a fault between two pedal sensor readings.
275 *
276 * Scales pedal_2 to match the range of pedal_1, then calculates the absolute difference.
277 * If the difference exceeds 10% of the full-scale value (i.e., >102.4 for a 10-bit ADC),
278 * the function considers this a fault and returns true. Otherwise, returns false.
279 * Also logs the readings and fault status for debugging.
280 *
281 * @param pedal_1 Raw value from pedal sensor 1 (uint16_t), intentionally casted to int16_t.
282 * @param pedal_2 Raw value from pedal sensor 2 (uint16_t), intentionally casted to int16_t.
283 * @param brake Raw value from brake sensor to pass to debug function
284 * @return true if the difference exceeds the threshold (fault detected), false otherwise.
285 */
286bool Pedal::check_pedal_fault(int16_t pedal_1, int16_t pedal_2, uint16_t brake)
287{
288
289 int16_t pedal_2_scaled = round((float)pedal_2 * 4.8 / 3.2); // round((float)pedal_2 * PEDAL_1_RANGE / PEDAL_2_RANGE);
290 DBG_THROTTLE_IN(pedal_1, pedal_2, pedal_2_scaled, brake);
291
292 int16_t delta = pedal_1 - pedal_2_scaled;
293 // if more than 10% difference between the two pedals, consider it a fault
294 if (delta > 102.4 || delta < -102.4) // 10% of 1024, rounded down to 102
295 {
296 DBG_THROTTLE_FAULT(DIFF_CONTINUING, delta);
297 return true;
298 }
299 return false;
300}
