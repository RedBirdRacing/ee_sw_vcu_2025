43/**
44 * @brief Updates pedal sensor readings, applies filtering, and checks for faults.
45 *
46 * Stores new pedal readings, applies an average filter, and updates car state.
47 * If a fault is detected between pedal sensors, sets fault flags and logs status.
48 *
49 * @param pedal_1 Raw value from pedal sensor 1.
50 * @param pedal_2 Raw value from pedal sensor 2.
51 * @param brake Raw value from brake sensor.
52 */
53void Pedal::update(uint16_t pedal_1, uint16_t pedal_2, uint16_t brake)
54{
55 // Record readings in buffer
56 pedal_value_1.push(pedal_1);
57 pedal_value_2.push(pedal_2);
58 brake_value.push(brake);
59
60 // Range of pedal 1 is APPS_PEDAL_1_RANGE, pedal 2 is APPS_PEDAL_2_RANGE;
61
62 // this is current taking the direct array the circular queue writes into. Bad idea to do anything other than a simple average
63 // if not using a linear filter, pass the pedalValue_1.getLinearBuffer() to the filter function to ensure the ordering is correct.
64 // can also consider injecting the filter into the queue if need
65 // depends on the hardware filter, reduce software filtering as much as possible
66
67 // currently a small average filter is good enough
68 car.adc.apps_5v = AVG_filter<uint16_t>(pedal_value_1.buffer, ADC_BUFFER_SIZE);
69 car.adc.apps_3v3 = AVG_filter<uint16_t>(pedal_value_2.buffer, ADC_BUFFER_SIZE);
70 car.adc.brake = AVG_filter<uint16_t>(brake_value.buffer, ADC_BUFFER_SIZE);
71
72 if (car.adc.apps_5v < apps_5v_min)
73 car.state.faults.bits.apps_5v_low = true;
74 if (car.adc.apps_5v > apps_5v_max)
75 car.state.faults.bits.apps_5v_high = true;
76 if (car.adc.apps_3v3 < apps_3v3_min)
77 car.state.faults.bits.apps_3v3_low = true;
78 if (car.adc.apps_3v3 > apps_3v3_max)
79 car.state.faults.bits.apps_3v3_high = true;
80 if (car.adc.brake < brake_min)
81 car.state.faults.bits.brake_low = true;
82 if (car.adc.brake > brake_max)
83 car.state.faults.bits.brake_high = true;
84
85 bool result = checkPedalFault();
86 if (result)
87 {
88 if (fault && car.millis - fault_start_millis > 100)
89 {
90 car.state.faults.bits.fault_exceeded = true; // Turning off the motor is achieved using another digital pin, not via canbus, but will still send 0 torque can frames
91
92 DBG_THROTTLE_FAULT(PedalFault::DiffExceed100ms);
93 return;
94 }
95 }
96
97 if (!checkPedalFault())
98 {
99 if (fault)
100 {
101 DBG_THROTTLE_FAULT(PedalFault::DiffResolved);
102 fault = false;
103 }
104 return;
105 }
106
107 // Pedal fault detected
108 if (fault) // if previously faulty
109 { // Previous scan is already faulty
110 if (car.millis - fault_start_millis > 100) // Faulty for more than 100 ms
111 {
112 car.state.faults.bits.fault_exceeded = true; // Turning off the motor is achieved using another digital pin, not via canbus, but will still send 0 torque can frames
113 car.state.status.bits.force_stop = true; // this force stop flag can only be reset by a power cycle
114
115 DBG_THROTTLE_FAULT(PedalFault::DiffExceed100ms);
116 return;
117 }
118 }
119 else // new fault detected
120 {
121 fault_start_millis = car.millis;
122 DBG_THROTTLE_FAULT(PedalFault::DiffStart);
123 }
124
125 fault = true;
126 return;
127}

129/**
130 * @brief Sends the appropriate CAN frame based on pedal and car state.
131 */
132void Pedal::sendFrame()
133{
134 if (car.state.status.bits.force_stop)
135 {
136 DBGLN_THROTTLE("Stopping motor: pedal fault");
137 motor_can.sendMessage(&stop_frame);
138 return;
139 }
140 if (car.state.status.bits.car_status != CarStatus::Drive)
141 {
142 switch (car.state.status.bits.car_status)
143 {
144 case CarStatus::Init:
145 DBGLN_THROTTLE("Stopping motor: in INIT.");
146 break;
147 case CarStatus::Startin:
148 DBGLN_THROTTLE("Stopping motor: in STARTIN.");
149 break;
150 case CarStatus::Bussin:
151 DBGLN_THROTTLE("Stopping motor: in BUSSIN.");
152 break;
153 default:
154 DBGLN_THROTTLE("Stopping motor: in UNKNOWN STATE.");
155 break;
156 }
157 motor_can.sendMessage(&stop_frame);
158 return;
159 }
160
161 int16_t torque_val = throttleTorqueMapping(pedal_final, car.adc.brake, FLIP_MOTOR_DIR);
162
163 torque_msg.can_id = MOTOR_COMMAND;
164 torque_msg.can_dlc = 3;
165 torque_msg.data[0] = 0x90; // 0x90 for torque, 0x31 for speed
166 torque_msg.data[1] = torque_val & 0xFF;
167 torque_msg.data[2] = (torque_val >> 8) & 0xFF;
168 motor_can.sendMessage(&torque_msg);
169 return;
170}

172/**
173 * @brief Maps the pedal ADC to a torque value, handling deadzones and faults.
174 *
175 * @param pedal Pedal ADC in the range of 0-1023.
176 * @param brake Brake ADC in the range of 0-1023.
177 * @param flip_dir Boolean indicating whether to flip the motor direction.
178 * @return Mapped torque value in the signed range of -TORQUE_MAX to TORQUE_MAX.
179 */
180constexpr int16_t Pedal::throttleTorqueMapping(const uint16_t pedal, const uint16_t brake, const bool flip_dir)
181{
182 if (brake > brake_map.start())
183 {
184 if (pedal > throttle_map.start())
185 {
186 car.state.status.bits.screenshot = true;
187 }
188 return brakeTorqueMapping(brake, flip_dir);
189 }
190 if (flip_dir)
191 {
192 return -throttle_map.interp(pedal);
193 }
194 return throttle_map.interp(pedal);
195}

209 if (flip_dir)

216/**
217 * @brief Checks for a fault between two pedal sensor readings.
218 *
219 * Scales pedal_2 to match the range of pedal_1, then calculates the absolute difference.
220 * If the difference exceeds 10% of the full-scale value (i.e., >102.4 for a 10-bit ADC),
221 * the function considers this a fault and returns true. Otherwise, returns false.
222 *
223 * @return true if the difference exceeds the threshold (fault detected), false otherwise.
224 */
225bool Pedal::checkPedalFault()
226{
227 car.digital.apps_3v3_scaled = car.adc.apps_3v3 * APPS_RATIO;
228
229 const int16_t delta = (int16_t)car.adc.apps_5v - (int16_t)car.digital.apps_3v3_scaled;
230 // if more than 10% difference between the two pedals, consider it a fault
231 if (delta > 102.4 || delta < -102.4) // 10% of 1024, rounded down to 102
232 {
233 car.state.faults.bits.fault_active = true;
234 DBG_THROTTLE_FAULT(PedalFault::DiffContinuing, delta);
235 return true;
236 }
237 car.state.faults.bits.fault_active = false;
238 return false;
239}
