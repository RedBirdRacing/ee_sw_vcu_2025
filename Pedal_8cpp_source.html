25/**
26 * @brief Constructor for the Pedal class.
27 * Initializes the pedal state. fault is set to true initially,
28 * so you must send update within 100ms of starting the car to clear it.
29 * Sends request to motor controller for cyclic RPM and error reads.
30 * @param motor_can_ Reference to the MCP2515 instance for motor CAN communication.
31 * @param car_ Reference to the CarState structure.
32 * @param pedal_final_ Reference to the pedal used as the final pedal value. Although not recommended, you can set another uint16 outside Pedal to be something like 0.3 APPS_1 + 0.7 APPS_2, then reference that here. If in future, this become a sustained need, should consider adding a function pointer to find the final pedal value to let Pedal class call it itself.
33 */
34Pedal::Pedal(MCP2515 &motor_can_, CarState &car_, uint16_t &pedal_final_)
35 : pedal_final(pedal_final_),
36 car(car_),
37 motor_can(motor_can_),
38 fault_start_millis(0),
39 last_motor_read_millis(0)
40{
41 // ask MCU to send motor rpm and error/warn signals
42 while (sendCyclicRead(SPEED_IST, RPM_PERIOD) != MCP2515::ERROR_OK)
43 ;
44 while (sendCyclicRead(WARN_ERR, ERR_PERIOD) != MCP2515::ERROR_OK)
45 ;
46 // set MCU CAN filter
47 while (motor_can.setFilter(MCP2515::RXF0, false, MOTOR_READ) != MCP2515::ERROR_OK)
48 ;
49}

51/**
52 * @brief Updates pedal sensor readings, applies filtering, and checks for faults.
53 *
54 * Stores new pedal readings, applies an average filter, and updates car state.
55 * If a fault is detected between pedal sensors, sets fault flags and logs status.
56 *
57 * @param pedal_1 Raw value from pedal sensor 1.
58 * @param pedal_2 Raw value from pedal sensor 2.
59 * @param brake Raw value from brake sensor.
60 */
61void Pedal::update(uint16_t pedal_1, uint16_t pedal_2, uint16_t brake)
62{
63 // Add new samples to the filters
64 pedal1_filter.addSample(pedal_1);
65 pedal2_filter.addSample(pedal_2);
66 brake_filter.addSample(brake);
67
68 if (pedal_1 < APPS_5V_MIN)
69 car.pedal.faults.bits.apps_5v_low = true;
70 if (pedal_1 > APPS_5V_MAX)
71 car.pedal.faults.bits.apps_5v_high = true;
72 if (pedal_2 < APPS_3V3_MIN)
73 car.pedal.faults.bits.apps_3v3_low = true;
74 if (pedal_2 > APPS_3V3_MAX)
75 car.pedal.faults.bits.apps_3v3_high = true;
76 if (brake < brake_min)
77 car.pedal.faults.bits.brake_low = true;
78 if (brake > brake_max)
79 car.pedal.faults.bits.brake_high = true;
80
81 if (checkPedalFault())
82 {
83 // fault now
84 if (car.pedal.faults.bits.fault_active)
85 {
86 // was faulty already, check time
87 if (car.millis - fault_start_millis > 100)
88 {
89 car.pedal.faults.bits.fault_exceeded = true;
90 car.pedal.status.bits.force_stop = true; // critical fault, force stop; since early return, need set here
91 DBG_THROTTLE_FAULT(PedalFault::DiffExceed100ms);
92 return;
93 }
94 else
95 {
96 DBG_THROTTLE_FAULT(PedalFault::DiffContinuing); // will be optimized out if the debug macro is off
97 }
98 }
99 else
100 {
101 // new fault
102 fault_start_millis = car.millis;
103 DBG_THROTTLE_FAULT(PedalFault::DiffStart);
104 }
105 car.pedal.faults.bits.fault_active = true;
106 }
107 else
108 {
109 // no fault
110 if (car.pedal.faults.bits.fault_active)
111 {
112 DBG_THROTTLE_FAULT(PedalFault::DiffResolved);
113 }
114 car.pedal.faults.bits.fault_active = false;
115 }
116
117 if (car.pedal.faults.byte & FAULT_CHECK_HEX)
118 {
119 car.pedal.status.bits.force_stop = true; // critical fault, force stop
120 }
121
122 return;
123}

125/**
126 * @brief Sends the appropriate CAN frame to the motor based on pedal and car state.
127 */
128void Pedal::sendFrame()
129{
130 // Update Telemetry struct
131 car.pedal.apps_5v = pedal1_filter.getFiltered();
132 car.pedal.apps_3v3 = pedal2_filter.getFiltered();
133 car.pedal.brake = brake_filter.getFiltered();
134
135 if (car.pedal.status.bits.force_stop)
136 {
137 DBGLN_THROTTLE("Stopping motor: pedal fault");
138 motor_can.sendMessage(&stop_frame);
139 return;
140 }
141 if (car.pedal.status.bits.car_status != CarStatus::Drive)
142 {
143 switch (car.pedal.status.bits.car_status)
144 {
145 case CarStatus::Init:
146 DBGLN_THROTTLE("Stopping motor: in INIT.");
147 break;
148 case CarStatus::Startin:
149 DBGLN_THROTTLE("Stopping motor: in STARTIN.");
150 break;
151 case CarStatus::Bussin:
152 DBGLN_THROTTLE("Stopping motor: in BUSSIN.");
153 break;
154 default:
155 DBGLN_THROTTLE("Stopping motor: in UNKNOWN STATE.");
156 break;
157 }
158 motor_can.sendMessage(&stop_frame);
159 return;
160 }
161
162 car.motor.torque_val = pedalTorqueMapping(pedal_final, car.pedal.brake, car.motor.motor_rpm, FLIP_MOTOR_DIR);
163
164 torque_msg.data[1] = car.motor.torque_val & 0xFF;
165 torque_msg.data[2] = (car.motor.torque_val >> 8) & 0xFF;
166 motor_can.sendMessage(&torque_msg);
167 return;
168}

170/**
171 * @brief Maps the pedal ADC to a torque value.
172 * If no braking requested, maps throttle normally.
173 * If braking requested and regen enabled,
174 * applies regen if motor RPM larger than minimum regen RPM,
175 * preventing reverse torque at low speeds.
176 * Regen is also disabled if motor rpm isn't read recently to prevent reverse power.
177 *
178 * @param pedal Pedal ADC in the range of 0-1023.
179 * @param brake Brake ADC in the range of 0-1023.
180 * @param motor_rpm Current motor RPM for regen logic, scaled to 0-32767.
181 * @param flip_dir Boolean indicating whether to flip the motor direction.
182 * @return Mapped torque value in the signed range of -TORQUE_MAX to TORQUE_MAX.
183 */
184constexpr int16_t Pedal::pedalTorqueMapping(const uint16_t pedal, const uint16_t brake, const int16_t motor_rpm, const bool flip_dir)
185{
186 if (REGEN_ENABLED && brake > BRAKE_MAP.start() && !car.pedal.status.bits.motor_no_read)
187 {
188 if (pedal > THROTTLE_MAP.start())
189 {
190 car.pedal.status.bits.screenshot = true;
191 // to ensure BSPD can be tested, skip regen if both throttle and brake pressed
192 }
193 else if (flip_dir)
194 {
195 if (motor_rpm < PedalConstants::MIN_REGEN_RPM_VAL)
196 return 0;
197 else
198 return -BRAKE_MAP.interp(brake);
199 }
200 else
201 {
202 if (motor_rpm > -PedalConstants::MIN_REGEN_RPM_VAL)
203 return 0;
204 else
205 return BRAKE_MAP.interp(brake);
206 }
207 }
208
209 if (flip_dir)
210 return -THROTTLE_MAP.interp(pedal);
211 else
212 return THROTTLE_MAP.interp(pedal);
213}

215/**
216 * @brief Checks for a fault between two pedal sensor readings.
217 *
218 * Scales pedal_2 to match the range of pedal_1, then calculates the absolute difference.
219 * If the difference exceeds 10% of the full-scale value (i.e., >102.4 for a 10-bit ADC),
220 * the function considers this a fault and returns true. Otherwise, returns false.
221 *
222 * @return true if the difference exceeds the threshold (fault detected), false otherwise.
223 */
224bool Pedal::checkPedalFault()
225{
226 const int16_t delta = (int16_t)car.pedal.apps_5v - (int16_t)APPS_3V3_SCALE_MAP.interp(car.pedal.apps_3v3);
227 constexpr int16_t MAX_DELTA = THROTTLE_MAP.range() / 10; /**< MAX_DELTA is floor of 10% of APPS_5V valid range, later comparison will give rounding room */
228 // if more than 10% difference between the two pedals, consider it a fault
229 if (delta > MAX_DELTA || delta < -MAX_DELTA)
230 {
231 DBG_THROTTLE_FAULT(PedalFault::DiffContinuing, delta);
232 return true;
233 }
234 return false;
235}

237/**
238 * @brief Sends a cyclic read request to the motor controller for speed (rpm).
239 * @param reg_id Register ID to read from the motor controller.
240 * @param read_period Period of reading motor data in ms.
241 * @return MCP2515::ERROR indicating success or failure of sending the message.
242 */
243MCP2515::ERROR Pedal::sendCyclicRead(uint8_t reg_id, uint8_t read_period)
244{
245 can_frame cyclic_request = {
246 MOTOR_SEND, /**< can_id */
247 3, /**< can_dlc */
248 REGID_READ, /**< data, register ID */
249 reg_id, /**< data, sub ID */
250 read_period /**< data, read period in ms */
251 };
252 return motor_can.sendMessage(&cyclic_request);
253}

255/**
256 * @brief Reads motor data from the CAN bus and updates the CarState.
257 */
258void Pedal::readMotor()
259{
260 can_frame rx_frame;
261 if (motor_can.readMessage(&rx_frame) == MCP2515::ERROR_OK)
262 {
263 if (rx_frame.can_id == MOTOR_READ && rx_frame.can_dlc > 3)
264 {
265 if (rx_frame.data[0] == SPEED_IST)
266 {
267 last_motor_read_millis = car.millis;
268 car.pedal.status.bits.motor_no_read = false;
269 car.motor.motor_rpm = static_cast<int16_t>(rx_frame.data[1] | (rx_frame.data[2] << 8));
270 return;
271 }
272 else if (rx_frame.data[0] == WARN_ERR)
273 {
274 car.motor.motor_error = static_cast<uint16_t>(rx_frame.data[1] | (rx_frame.data[2] << 8));
275 car.motor.motor_warn = static_cast<uint16_t>(rx_frame.data[3] | (rx_frame.data[4] << 8));
276 return;
277 }
278 }
279 }
280 if (car.millis - last_motor_read_millis > MAX_MOTOR_READ_MILLIS)
281 {
282 car.pedal.status.bits.motor_no_read = true;
283 DBG_THROTTLE("No motor read for over 100 ms, disabling regen");
284 }
285 return;
286}
