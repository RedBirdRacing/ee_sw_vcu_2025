137 // pedal is still being updated, data can still be gathered and sent through CAN/serial
138 }
139
140 switch (car.state.status.bits.car_status)
141 {
142 case DRIVE:
143 // Pedal update
144 // Send pedal value through canbus
145 // already scheduled
146 return; // no need logic to check if pedal on, car started
147
148 // do not return here if not in DRIVE mode, else can't detect pedal being on while starting
149 case INIT:
150 DBGLN_THROTTLE("Stopping motor: INIT.");
151
152 if (digitalRead(DRIVE_MODE_BTN) == BUTTON_ACTIVE && brake_pressed)
153 {
154 car.state.status.bits.car_status = STARTIN;
155 car.status_millis = car.millis;
156
157 scheduler.addTask(MCP_BMS, scheduler_bms, 5); // check for HV ready in STARTIN
158 }
159 break;
160
161 case STARTIN:
162 DBGLN_THROTTLE("Stopping motor: STARTIN.");
163
164 if (digitalRead(DRIVE_MODE_BTN) != BUTTON_ACTIVE || !brake_pressed)
165 {
166 car.state.status.bits.car_status = INIT;
167 car.status_millis = car.millis; // safety
168 scheduler.removeTask(MCP_BMS, scheduler_bms); // stop checking BMS HV ready since return to INIT
169 }
170 else if (car.millis - car.status_millis >= STARTING_MILLIS)
171 {
172 scheduler.removeTask(MCP_BMS, scheduler_bms); // stop checking BMS HV ready, either started or return to INIT
173 if (bms.hvReady()) // if HV not started, return to INIT
174 {
175 car.state.status.bits.car_status = INIT;
176 car.status_millis = car.millis; // safety
177 break;
178 }
179 if (car.millis - car.status_millis >= BMS_MILLIS)
180 {
181 car.state.status.bits.car_status = BUSSIN;
182 car.status_millis = car.millis; // safety
183 break;
184 }
185 }
186 break;
187
188 case BUSSIN:
189 digitalWrite(BUZZER, HIGH); // Turn on buzzer
190
191 if (car.millis - car.status_millis >= BUSSIN_MILLIS)
192 {
193 digitalWrite(BUZZER, LOW);
194 digitalWrite(FRG, HIGH);
195 car.state.status.bits.car_status = DRIVE;
196 }
197 break;
198
199 default:
200 // unreachable, reset to INIT
201 car.state.status.bits.car_status = INIT;
202 car.status_millis = car.millis;
203 break;
204 }

207 if (pedal.pedal_final > apps_final_min) // if pedal pressed while not in DRIVE, reset to INIT
